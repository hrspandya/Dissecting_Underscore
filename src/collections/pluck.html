	    	//Does Shallow check 
    		function pluck(list, propertyName){
    			var result = [];
    			//loop through list
    			for(var i = 0; i < list.length; i++){
    				
    				var listNode = list[i];
    				var propertyObj = listNode[propertyName];

    				//make sure its object
    				if(listNode.constructor === Object){

    					//find propertyName
    					if(propertyObj !== void 0){
    						result.push(list[i][propertyName]);
    					}

    				}
    			}

    			return result;
    		}

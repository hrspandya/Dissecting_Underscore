			//passing array/array of objects , function, and Context
			function groupBy(list, func, context){								
				
				var result, resultObject = {};

				//Make sure list is an Array
				if(list.constructor === Array){
					
					//Loop through array
					for(var i = 0; i < list.length; i++){
						
						//if Function is passed
						if(func.constructor === Function){
							
							//getting result from calling the function
							result = func(list[i]);

						}else if(func.constructor === String){
							//is String is passed

							result = list[i][func];
						}

						//Since we need to groupBy , lets save this result with Object key and value							
						if(resultObject[result] === void 0){
							resultObject[result] = [];
						}

						resultObject[result].push(list[i]);
					}

				}

				return resultObject;
			}


			function callbackFunc(num){				
				return Math.floor(num);

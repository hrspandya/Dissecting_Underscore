My Solution: 
	    	//passing array/object/string , method String, and arguments
            function invoke(array, methodName){     
                var args = [];
                var result, func;

                //Check arguments are more than 2, if so we need to capture arguments and pass it along in the method
                if(arguments.length > 2){
                    for(var i = 2; i < arguments.length; i++){
                        args.push(arguments[i]);
                    }
                }               
                
                func = array[methodName]; 	//array.methodName //array.sort returns function

                result = func.apply(array, args);

                return result;
            }



            //Calling Method and show output by passing Array                               
            $(".sortString output").append(invoke(["b", "c", "a", "d"], "sort").toString());


            //Rememeber that javscript sort method by default does String sort only, so passing custom function to sort in asc order
            $(".sortNumber output").append(invoke([100, 20, 3, 9, 33], "sort", function(a, b){return a - b;}).toString());


            //Calling Method and Show Output by passing Object                      
            $(".outputObject output").append(invoke("66635435345", "slice", 5).toString());

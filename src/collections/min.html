           var stooges = [
                {name: 'moe', age: 40, test : [{name : 'sam'}]}, 
                {name: 'larry', age: 50}, 
                {name: 'curly', age: 60}
            ];

            var numbers = [2, 4, 10, 22, 11, 19];

            
            //Does Shallow check 
            function min(list, func, context){
                var value;
                var result;
                //loop through list
                for(var i = 0; i < list.length; i++){
                    debugger;
                    if(func !== void 0 && func.constructor === Function){
                        // if function is passed 
                        
                        //Since Context is optional can be undefined, and so checking for undefined,                    
                        if(context == void 0){
                            var returnedValue;

                            //If undefined then dont pass context
                            //Get value back from function call
                            returnedValue = func.call(list[i]);

                        }else{
                            //for each array value call the function passed in the argument with given context, 
                            //Since we need to run this function based on context , we need to use call or apply
                            //For this example i am only using call
                            returnedValue = func.call(context);
                        }

                        //compare which one is smaller
                        if(value === void 0 || value > returnedValue){
                            value = returnedValue;
                            result = list[i];   //saving result
                        }

                    }else{
                        //compare which one is smaller
                        if(value === void 0 || value > list[i]){
                            value = list[i];
                            result = list[i]    //saving result
                        }    
                    }
                    
                }

                return result;
            }

            


            function functionCallback(){
                return this["age"];
            }


            //Calling Method and show output by passing Array                               
            $(".outputArray output").append(JSON.stringify(min(numbers)).toString());


            //Calling Method and Show Output by passing Array of Objects
            $(".outputObject output").append(JSON.stringify(min(stooges, functionCallback)).toString());

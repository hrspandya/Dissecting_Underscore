            //Creates a version of the function that can be called no more than count times. 
            //The result of the last function call is memoized and returned when count has been reached.
            function before(count, func){                
                
                var times = count;
                var memo;

                return function(){
                    
                    if(times-- > 0){
                        memo = func.apply(this, arguments);
                        return memo;
                    }else{
                        return memo;
                    }

                }

            }


            var initialize = before(3, function(arg){
                return arg;
            });
            
            $(".result1 output").append(" " + initialize(1) + " ");
            $(".result1 output").append(" " + initialize(2) + " ");
            $(".result1 output").append(" " + initialize(3) + " ");
            $(".result1 output").append(" " + initialize(4) + " ");

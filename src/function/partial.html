            function partial(func){

                //get arguments 
                var argsArray = Array.prototype.slice.call(arguments, 1);                
                var _passed = false;
                var arg1;               

                if(argsArray.length > 0){
                    if(argsArray[0] == "_"){
                        _passed = true; 
                        arg1 = argsArray[1];
                    }else{
                        arg1 = argsArray[0];
                        _passed = false;
                    }
                }else{
                    return undefined;
                }

                return function(arg){
                    var arg2 = arg;
                    
                    if(_passed){
                        return func.apply(null, [arg2, arg1]);
                    }else{
                        return func.apply(null, [arg1, arg2]);
                    }                    
                };
            }


            var subtract = function(a, b) { 
                return b - a; 
            };

            sub5 =  partial(subtract, 5);
            $(".result1 output").append(sub5(20));
            

            // Using a placeholder
            subFrom20 = partial(subtract, "_", 20);
            $(".result2 output").append(subFrom20(5));

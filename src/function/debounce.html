            //debounce means everytime function is called we reset the timeout and call function only after timeout has elapsed            
            function debounce(func, wait, isImmediate){                
                
                var timer, busy = false;

                function later(){
                    var args = arguments;

                    //leading edge, call function first and then wait for timeout
                    if(isImmediate){
                        
                        //by default busy is false, if busy return , ignore                        
                        if(busy){
                            return;
                        }

                        //since we are about to call function so make it busy
                        busy = true;
                        func.apply(this, args);

                        //now keep it busy until timeout
                        setTimeout(function(){
                            //once timeout is complete busy is false again
                            busy = false;
                        }, wait);

                    }else{

                        //in the case of trailing, we just need to create timer, clear timeout everytime function gets called
                        clearTimeout(timer);

                        //start timer, if timeout successfully only then call function
                        timer = setTimeout(function(){
                            func.apply(this, args);
                        }, wait);
                    }
                    
                }   

                return later;         
            }

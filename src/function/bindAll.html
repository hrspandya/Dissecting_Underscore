            function bind(func, context){
                'use strict'                                

                //convert arguments into an array and slice on index 2
                //arguments is not an array, we need to explicitly convert arguments into and array 
                var argsArray = Array.prototype.slice.call(arguments, 2);

                if(func.constructor !== Function){
                    return void 0;
                }
                
                //if ar
                if(argsArray.length > 0){
                    return function(){
                        return func.apply(context, argsArray);
                    };
                }else{
                    return function(){
                        return func.call(context);
                    };
                }                

            }


            function bindAll(context){
                'use strict'                                

                //convert arguments into an array and slice on index 1
                //arguments is not an array, we need to explicitly convert arguments into and array 
                var argsArray = Array.prototype.slice.call(arguments, 1);

                if(argsArray.length == 0){
                    return void 0;
                }else if(argsArray.length >= 1){
                    for(var i = 0; i < argsArray.length; i++){

                        //Call bind method here, which we discussed before, 
                        //bind function returns a function which gets assigned back to original methods but with given context
                        context[argsArray[i]] = bind(context[argsArray[i]], context);                        
                    }
                }

            }

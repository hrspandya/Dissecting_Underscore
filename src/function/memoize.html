           function memoize(func){
                            
                var memo = {};               

                return function(){
                    
                    var argArray = Array.prototype.slice.call(arguments);

                    if(argArray in memo){
                        return memo[argArray];
                    }else{
                        memo[argArray] = func.apply(this, argArray);
                        return memo[argArray];
                    }

                }
                
            }


            var fibonacci = function(n) {
                return n < 2 ? n: fibonacci(n - 1) + fibonacci(n - 2);
            };

            var fib = memoize(fibonacci);
            
            $(".result1 output").append(fib(20));
                      
            $(".result2 output").append(fib(20));

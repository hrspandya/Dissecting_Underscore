            //zip values in array based on index    
            function object(list){
                'use strict'

                var resultArray = [];         
                var arrayLength;
                var map = {};                       
            
                if(arguments !== void 0 && arguments.length == 1){
                    return list;
                }else if(arguments !== void 0 && arguments.length > 1){

                    //loop through array
                    for(var i = 0; i < arguments.length; i++){
                        
                        // find array length and save it for later use
                        if(arrayLength === void 0){
                            arrayLength = arguments[i].length;
                        }                        

                        if(arguments[i].length !== arrayLength){
                            return void 0; //arrays dont have same lengths
                        }

                        //loop though each array and add first array in the map
                        for(var j = 0; j < arguments[i].length; j++){
                            if(map[j] === void 0){
                                map[j] = [arguments[i][j]];
                            }else{
                                map[j].push(arguments[i][j]);
                            }
                        }

                    }


                    for(var index in map){
                        if(map.hasOwnProperty(index)){
                            resultArray.push(map[index]);
                        }                        
                    }
                }



                return resultArray;

            }

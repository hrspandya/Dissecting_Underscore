My Solution: 
	    	
            function difference(list){

                var resultArray = [];                                
                var argsMap = {};
                var concatedArrays = [];
                
                //loop through arguments but skip first, since list is our input array
                for(var i = 1; i < arguments.length; i++){
                    concatedArrays = concatedArrays.concat(arguments[i]);
                }

                //loop through concated arrays and put them in the map
                for(var j = 0; j < concatedArrays.length; j++){  
                    argsMap[concatedArrays[j]] = concatedArrays[j];
                }

                //Now loop through your list and find difference
                for(var k = 0; k < list.length; k++){
                    if(list[k] !== argsMap[list[k]]){
                        resultArray.push(list[k]);
                    }
                }

                return resultArray;

            }

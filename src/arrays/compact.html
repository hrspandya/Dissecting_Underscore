            function compact(list){

                var resultArray = [];

                if(list.constructor === Array){

                    for(var i = 0; i < list.length; i++){
                        
                        if(list[i] == void 0 
                            || list[i] == false 
                            || list[i] == null
                            || list[i] == 0 
                            || list[i] == ""
                            || isNaN(list[i])){

                            //Do nothing
                        }else{
                            resultArray.push(list[i]);
                        }

                    }

                }

                return resultArray;

            }

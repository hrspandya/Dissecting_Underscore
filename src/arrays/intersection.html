_.intersection(*arrays) 
    	
    		_.intersection([1, 2, 3], [101, 2, 1, 10], [2, 1]);
                => [1, 2]
My Solution: 
	    	  
            function intersection(list){

                var resultArray = [];                                
                var argsMap = {};
                
                //Make sure arguments length is more than 0, otherwise return []                                  
                if(arguments.length > 0){

                    for(var i = 0; i < list.length; i++){

                        //Loop through arguments array and make sure each arg is an array
                        if(arguments[i].constructor === Array){

                            for(var k = 0; k < arguments[i].length; k++){

                                //since we are looking for intersection, we need to make sure that count reaches to (arguemnts.length - 1)
                                //which means every array has the value we are looking for 
                                if(argsMap[arguments[i][k]] === arguments.length - 1){
                                    //if all arrays have the value than add it to result Array
                                    resultArray.push(arguments[i][k]);
                                }
                                
                                //when map value is undefined make it 1
                                if(argsMap[arguments[i][k]] === void 0){
                                    argsMap[arguments[i][k]] = 1
                                }else{
                                    //otherwise increment
                                    argsMap[arguments[i][k]] = argsMap[arguments[i][k]] + 1;
                                }

                            }
                        }
                    }

                }else{

                    return list;

                }

                return resultArray;

            }

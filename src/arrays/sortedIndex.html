            function sortedIndex(list, n){
                var lowerbound = 0;
                var upperbound = list.length;
                
                while(lowerbound < upperbound){
                                
                    var mid = Math.floor((lowerbound + upperbound)/2);

                    //value already exist no need to add it again
                    if(list[mid] === n){
                        return - 1;
                    }else if(n < list[mid]){
                        upperbound = mid - 1;
                    }else if(n > list[mid]){   
                        lowerbound = mid + 1;
                    }    

                }

                return lowerbound;

            }

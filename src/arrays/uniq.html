            function uniq(list){
                'use strict'

                var resultArray = [];                                
                var argsMap = {};
                //check for array
                if(list.constructor === Array){

                    //loop through array
                    for(var i = 0; i < list.length; i++){                        

                        //check if map has this value
                        if(argsMap[list[i]] === void 0){
                            //if map does not have the value add and push to the result array
                            argsMap[list[i]] = list[i];
                            resultArray.push(list[i]);
                        }
                    }
                }                

                //return result
                return resultArray;

            }


            //Calling Method and show output by passing Array                               
            $(".result1 output").append(JSON.stringify(uniq([1, 2, 1, 4, 1, 3])).toString());
